/* Commands to set up the necessary database and tables */

CREATE DATABASE sporks50_char;
USE sporks50_char;
CREATE TABLE IF NOT EXISTS cp_names ( ID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT, URI TEXT, Page FLOAT(8.2) );
CREATE TABLE IF NOT EXISTS cp_contents ( ID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT, Page FLOAT(8.2), Content MEDIUMTEXT );
CREATE TABLE IF NOT EXISTS cp_items ( ID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT, Name TEXT, Page FLOAT(8.2), Description LONGTEXT );
CREATE TABLE IF NOT EXISTS cp_styles ( ID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT, Type TINYTEXT, Name TINYTEXT, Page FLOAT(8.2), Description LONGTEXT );
CREATE TABLE IF NOT EXISTS cp_types ( ID MEDIUMINT PRIMARY KEY NOT NULL AUTO_INCREMENT, Name TINYTEXT, Type TINYTEXT );
